def count_images_per_class(dataset_directory):
    class_image_counts = {}

    for category_label in os.listdir(dataset_directory):
        category_directory = os.path.join(dataset_directory, category_label)

        if os.path.isdir(category_directory):
            image_file_list = [
                f for f in os.listdir(category_directory)
                if f.lower().endswith(('.png', '.jpg', '.jpeg'))
            ]

            class_image_counts[category_label] = len(image_file_list)

    return class_image_counts

def delete_corrupted_images(dataset_directory):

def get_image_dimensions(dataset_root):
    image_widths, image_heights = [], []
    processed_image_count = 0
    for folder, _, files in os.walk(dataset_root):
        for file_name in files:
            if file_name.lower().endswith(('.jpg', '.jpeg', '.png')):
                current_image_path = os.path.join(folder, file_name)
                try:
                    with Image.open(current_image_path) as image_object:
                        width_values, height_values = image_object.size
                        image_widths.append(width_values)
                        image_heights.append(height_values)
                    processed_image_count += 1
                except Exception as e:
                    print(f"Error processing file {file_name}: {e}")

    if processed_image_count == 0:
        print("No images were processed.")
        return np.array([]), np.array([])

    return np.array(image_widths), np.array(image_heights)
